#define Degree 255
#define Led 11
#define Button 8
#define Hotpack 5
void setup(){
  pinMode(Led,OUTPUT);
  pinMode(Button,INPUT);
}
void loop(){
  
  if(digitalRead(Button)==HIGH)
  {
  analogWrite(Led,Degree);
  
  
  digitalWrite(Led,LOW);
  delay(1000);
  analogWrite(Hotpack,Degree);
  delay(100000);
  
  digitalWrite(Hotpack,LOW);
  }
  
}
